# 광고 분석 주간 쿼리 가이드 (기간조회 버전)

## 7. 실제 활용 예시 (Redash 기간 파라미터 활용)

### 7.1 기간 파라미터를 활용한 개선된 쿼리
Redash의 Date 파라미터 2개(시작일/종료일)를 사용하여 기간별 분석을 지원합니다.

#### 주간 광고주별 성과 (기간조회 버전)
```sql
-- 광고주별 기간 성과 - 기간별 집계
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as days_count
),
period_summary AS (
    SELECT 
        advertiser_id,
        advertiser_name,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as period_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as period_cost,
        COUNT(*) as period_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as period_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as period_conversions,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY advertiser_id, advertiser_name
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    days_count as "기간_일수",
    advertiser_id as "광고주ID",
    advertiser_name as "광고주명",
    active_days as "활동_일수",
    period_revenue as "기간_매출",
    period_cost as "기간_광고비",
    period_impressions as "기간_노출수",
    period_clicks as "기간_클릭수",
    period_conversions as "기간_전환수",
    CASE WHEN period_cost > 0 
         THEN ROUND(period_revenue / period_cost, 2)
         ELSE 0 END as "기간_ROAS",
    ROUND(period_revenue / days_count, 0) as "일평균_매출",
    ROUND(period_cost / days_count, 0) as "일평균_광고비"
FROM period_summary
CROSS JOIN date_params
ORDER BY period_revenue DESC
```

#### 기간 비교 분석 (기간조회 버전)
```sql
-- 두 기간 비교 - 사용자 지정 기간 간 비교
WITH period1 AS (
    SELECT 
        DATE('{{ start_date }}') as p1_start,
        DATE('{{ end_date }}') as p1_end,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as p1_days
),
period2 AS (
    SELECT 
        DATE('{{ compare_start_date }}') as p2_start,
        DATE('{{ compare_end_date }}') as p2_end,
        DATE_DIFF('day', DATE('{{ compare_start_date }}'), DATE('{{ compare_end_date }}')) + 1 as p2_days
),
period1_data AS (
    SELECT 
        '기간1' as "구분",
        date_format(p1_start, '%Y-%m-%d') || ' ~ ' || date_format(p1_end, '%Y-%m-%d') as "기간",
        p1_days as "일수",
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
        COUNT(*) as "노출수",
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수"
    FROM ad_combined_log_summary
    CROSS JOIN period1
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN p1_start AND p1_end
    GROUP BY 1, 2, 3
),
period2_data AS (
    SELECT 
        '기간2' as "구분",
        date_format(p2_start, '%Y-%m-%d') || ' ~ ' || date_format(p2_end, '%Y-%m-%d') as "기간",
        p2_days as "일수",
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
        COUNT(*) as "노출수",
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수"
    FROM ad_combined_log_summary
    CROSS JOIN period2
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN p2_start AND p2_end
    GROUP BY 1, 2, 3
)
SELECT 
    "구분",
    "기간",
    "일수",
    "매출",
    "광고비",
    "노출수",
    "클릭수",
    "전환수",
    CASE WHEN "광고비" > 0 THEN ROUND("매출" / "광고비", 2) ELSE 0 END as "ROAS",
    CASE WHEN "노출수" > 0 
         THEN ROUND(CAST("클릭수" AS DOUBLE) / "노출수" * 100, 2) 
         ELSE 0 END as "CTR",
    ROUND("매출" / "일수", 0) as "일평균_매출"
FROM (
    SELECT * FROM period1_data
    UNION ALL
    SELECT * FROM period2_data
)
```

### 7.2 Redash 파라미터 설정
- **start_date**: Date 타입, 기간 시작일
- **end_date**: Date 타입, 기간 종료일  
- **compare_start_date**: Date 타입, 비교 기간 시작일 (비교 분석용)
- **compare_end_date**: Date 타입, 비교 기간 종료일 (비교 분석용)

#### 기간 내 일별 성과 추이 (기간조회 버전)
```sql
-- 기간 내 일별 성과 추이 - 날짜별 트렌드 확인
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
daily_performance AS (
    SELECT 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
        day_of_week(date_parse(concat(CAST(year AS VARCHAR), '-', 
                               LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                               LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as dow,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as daily_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as daily_cost,
        COUNT(*) as daily_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as daily_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as daily_conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY 1, 2
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    date_format(date, '%Y-%m-%d') as "날짜",
    CASE dow
        WHEN 1 THEN '일요일'
        WHEN 2 THEN '월요일'
        WHEN 3 THEN '화요일'
        WHEN 4 THEN '수요일'
        WHEN 5 THEN '목요일'
        WHEN 6 THEN '금요일'
        WHEN 7 THEN '토요일'
    END as "요일",
    daily_revenue as "일별_매출",
    daily_cost as "일별_광고비",
    daily_impressions as "일별_노출수",
    daily_clicks as "일별_클릭수",
    daily_conversions as "일별_전환수",
    CASE WHEN daily_cost > 0 
         THEN ROUND(daily_revenue / daily_cost, 2)
         ELSE 0 END as "일별_ROAS",
    CASE WHEN daily_impressions > 0 
         THEN ROUND(CAST(daily_clicks AS DOUBLE) / daily_impressions * 100, 2) 
         ELSE 0 END as "일별_CTR"
FROM daily_performance dp
CROSS JOIN date_params
ORDER BY date
```

#### 기간 캠페인별 ROAS 순위 (기간조회 버전)
```sql
-- 기간 캠페인별 ROAS TOP 10 - 날짜 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
campaign_period AS (
    SELECT 
        campaign_id,
        campaign_name,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
        AND cost_per_impression + CASE WHEN is_click THEN cost_per_click ELSE 0 END > 0
    GROUP BY campaign_id, campaign_name
    HAVING SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    ROW_NUMBER() OVER (ORDER BY revenue / cost DESC) as "순위",
    CONCAT(SUBSTR(campaign_id, 1, 4), '..', SUBSTR(campaign_id, -4)) as "캠페인ID",
    campaign_name as "캠페인명",
    revenue as "매출",
    cost as "광고비",
    ROUND(revenue / cost, 2) as "ROAS",
    impressions as "노출수",
    clicks as "클릭수",
    conversions as "전환수",
    CASE WHEN impressions > 0 
         THEN ROUND(CAST(clicks AS DOUBLE) / impressions * 100, 2) 
         ELSE 0 END as "CTR",
    CASE WHEN clicks > 0 
         THEN ROUND(CAST(conversions AS DOUBLE) / clicks * 100, 2) 
         ELSE 0 END as "CVR"
FROM campaign_period cw
CROSS JOIN date_params
ORDER BY revenue / cost DESC
LIMIT 10
```

#### 기간 신규 광고주 성과 (기간조회 버전)
```sql
-- 기간 신규 광고주 성과 분석 - 기간 내 처음 시작한 광고주
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
before_period_advertisers AS (
    SELECT DISTINCT advertiser_id
    FROM ad_combined_log_summary
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        < (SELECT start_date FROM date_params)
),
new_advertisers AS (
    SELECT 
        ads.advertiser_id,
        ads.advertiser_name,
        MIN(date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as first_ad_date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary ads
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
        AND ads.advertiser_id NOT IN (SELECT advertiser_id FROM before_period_advertisers)
    GROUP BY ads.advertiser_id, ads.advertiser_name
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    advertiser_id as "광고주ID",
    advertiser_name as "광고주명",
    date_format(first_ad_date, '%Y-%m-%d') as "첫광고일",
    revenue as "기간_매출",
    cost as "기간_광고비",
    impressions as "기간_노출수",
    clicks as "기간_클릭수",
    conversions as "기간_전환수",
    CASE WHEN cost > 0 
         THEN ROUND(revenue / cost, 2)
         ELSE 0 END as "기간_ROAS"
FROM new_advertisers na
CROSS JOIN date_params
ORDER BY revenue DESC
```

#### 기간 피크타임 분석 (기간조회 버전)
```sql
-- 기간 시간대별 피크타임 분석 - 시간별 평균 성과
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
hourly_stats AS (
    SELECT 
        hour,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as days_active,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as hourly_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as hourly_cost,
        COUNT(*) as hourly_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as hourly_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as hourly_conversions,
        COUNT(DISTINCT user_id) as unique_users
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY hour
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    hour as "시간대",
    CASE 
        WHEN hour BETWEEN 0 AND 5 THEN '새벽(0-6시)'
        WHEN hour BETWEEN 6 AND 11 THEN '오전(6-12시)'
        WHEN hour BETWEEN 12 AND 17 THEN '오후(12-18시)'
        WHEN hour BETWEEN 18 AND 23 THEN '저녁(18-24시)'
    END as "시간대_구분",
    days_active as "활동일수",
    hourly_revenue as "시간대_매출",
    hourly_cost as "시간대_광고비",
    hourly_impressions as "시간대_노출수",
    hourly_clicks as "시간대_클릭수",
    hourly_conversions as "시간대_전환수",
    unique_users as "순사용자수",
    ROUND(hourly_revenue / days_active, 0) as "일평균_매출",
    CASE WHEN hourly_cost > 0 
         THEN ROUND(hourly_revenue / hourly_cost, 2)
         ELSE 0 END as "시간대_ROAS",
    CASE WHEN hourly_impressions > 0 
         THEN ROUND(CAST(hourly_clicks AS DOUBLE) / hourly_impressions * 100, 2) 
         ELSE 0 END as "시간대_CTR"
FROM hourly_stats hs
CROSS JOIN date_params
ORDER BY hourly_revenue DESC
```

#### 기간 광고 피로도 분석 (기간조회 버전)
```sql
-- 기간 사용자 광고 노출 빈도 및 피로도 분석  
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
user_exposure AS (
    SELECT 
        user_id,
        COUNT(*) as total_exposures,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as total_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as total_conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY user_id
),
exposure_buckets AS (
    SELECT 
        CASE 
            WHEN total_exposures <= 10 THEN '1-10회'
            WHEN total_exposures <= 20 THEN '11-20회'
            WHEN total_exposures <= 30 THEN '21-30회'
            WHEN total_exposures <= 50 THEN '31-50회'
            WHEN total_exposures <= 100 THEN '51-100회'
            ELSE '100회 초과'
        END as exposure_bucket,
        COUNT(*) as user_count,
        SUM(total_exposures) as total_exposures,
        SUM(total_clicks) as total_clicks,
        SUM(total_conversions) as total_conversions
    FROM user_exposure
    GROUP BY 1
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    exposure_bucket as "노출빈도_구간",
    user_count as "사용자수",
    total_exposures as "총노출수",
    total_clicks as "총클릭수",
    total_conversions as "총전환수",
    ROUND(CAST(total_exposures AS DOUBLE) / user_count, 1) as "사용자당_평균노출",
    CASE WHEN total_exposures > 0 
         THEN ROUND(CAST(total_clicks AS DOUBLE) / total_exposures * 100, 2) 
         ELSE 0 END as "CTR",
    CASE WHEN total_clicks > 0 
         THEN ROUND(CAST(total_conversions AS DOUBLE) / total_clicks * 100, 2) 
         ELSE 0 END as "CVR",
    ROUND(100.0 * user_count / SUM(user_count) OVER (), 1) as "사용자_비율"
FROM exposure_buckets eb
CROSS JOIN date_params
ORDER BY 
    CASE exposure_bucket
        WHEN '1-10회' THEN 1
        WHEN '11-20회' THEN 2
        WHEN '21-30회' THEN 3
        WHEN '31-50회' THEN 4
        WHEN '51-100회' THEN 5
        ELSE 6
    END
```

#### 기간 예산 소진율 분석 (기간조회 버전)
```sql
-- 기간 광고주별 예산 소진율 분석 - 기간 예산 한도 대비
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as period_days
),
advertiser_budget AS (
    SELECT 
        advertiser_id,
        advertiser_name,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as spent_budget,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY advertiser_id, advertiser_name
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    advertiser_id as "광고주ID",
    advertiser_name as "광고주명",
    spent_budget as "소진_예산",
    {{ daily_budget_limit|1000000 }} * period_days as "기간_예산한도",
    ROUND(spent_budget / ({{ daily_budget_limit|1000000 }} * period_days) * 100, 1) as "예산_소진율",
    active_days || '/' || period_days as "활동일수",
    ROUND(spent_budget / active_days, 0) as "일평균_소진액",
    revenue as "매출",
    CASE WHEN spent_budget > 0 
         THEN ROUND(revenue / spent_budget, 2)
         ELSE 0 END as "ROAS",
    CASE 
        WHEN spent_budget / ({{ daily_budget_limit|1000000 }} * period_days) >= 0.95 THEN '예산소진임박'
        WHEN spent_budget / ({{ daily_budget_limit|1000000 }} * period_days) >= 0.8 THEN '정상소진'
        WHEN spent_budget / ({{ daily_budget_limit|1000000 }} * period_days) >= 0.5 THEN '소진중간'
        ELSE '소진미달'
    END as "소진_상태"
FROM advertiser_budget ab
CROSS JOIN date_params
WHERE spent_budget > 0
ORDER BY spent_budget DESC
LIMIT 20
```

### 7.3 활용 시나리오
- **자유로운 기간 설정**: 7일, 14일, 30일 등 원하는 기간으로 분석
- **캠페인 기간 분석**: 특정 캠페인 진행 기간만 선택하여 성과 측정
- **이벤트 효과 분석**: 이벤트 기간 전후로 나누어 비교 분석
- **계절성 분석**: 같은 기간의 작년 데이터와 비교
- **A/B 테스트**: 서로 다른 기간의 테스트 결과 비교
- **예산 집행 관리**: 원하는 기간의 예산 소진율 확인
- **신규 광고주 추적**: 특정 기간에 시작한 광고주의 초기 성과 분석
