# CAPA 광고 분석 통합 대시보드 가이드

## 목적
일간, 주간, 월간 핵심 지표를 하나의 대시보드에서 모니터링하여 빠른 의사결정 지원

## 날짜 파라미터 설정
```yaml
# Redash Query Parameter (필수)
current_date:
  type: Date
  default: today  # 기본값: 오늘 날짜
  description: "분석 기준 날짜 (일반적으로 오늘 또는 어제)"
```

## 대시보드 구성 (총 10개 지표)

### 📊 일간 지표 (5개) - 실시간 운영 모니터링

#### 1. 오늘의 실시간 성과 (메인 지표)
```sql
-- 오늘 실시간 매출/광고비/ROAS - 매시간 업데이트
-- ⚠️ 주의: ad_combined_log 테이블은 conversion 데이터가 없으므로 ROAS 계산 불가
-- 실시간 conversion은 별도 조인이 필요하거나 ad_combined_log_summary(일별) 사용
WITH today_realtime AS (
    SELECT 
        COUNT(DISTINCT advertiser_id) as active_advertisers,
        -- conversion 데이터 없음 - 0으로 표시
        0 as today_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as today_cost,
        COUNT(*) as today_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as today_clicks,
        -- conversion 데이터 없음
        0 as today_conversions
    FROM ad_combined_log  -- 시간별 테이블 사용 (hour 파티션 있음)
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') = DATE '{{current_date}}'
)
SELECT 
    DATE_FORMAT(DATE '{{current_date}}', '%Y-%m-%d') as "날짜",
    DATE_FORMAT(CAST('{{current_date}}' AS TIMESTAMP), '%H:%i') as "업데이트_시각",
    active_advertisers as "활성_광고주",
    ROUND(today_cost, 0) as "오늘_광고비",
    today_clicks as "클릭수",
    today_impressions as "노출수",
    CASE WHEN today_impressions > 0 
         THEN ROUND(CAST(today_clicks AS DOUBLE) / today_impressions * 100, 2)
         ELSE 0 END as "CTR%",
    '⚠️ conversion 데이터는 일별 집계(ad_combined_log_summary) 사용 필요' as "참고"
FROM today_realtime
```

#### 2. 시간대별 피크타임 현황 (최근 24시간)
```sql
-- 오늘 시간대별 성과 히트맵
WITH hourly_today AS (
    SELECT 
        CAST(hour AS INTEGER) as hour,
        -- conversion 데이터 없음
        0 as hourly_revenue,
        COUNT(*) as hourly_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as hourly_clicks,
        -- conversion 데이터 없음
        0 as hourly_conversions,
        COUNT(DISTINCT user_id) as unique_users
    FROM ad_combined_log  -- 시간별 테이블 사용
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') = DATE '{{current_date}}'
    GROUP BY CAST(hour AS INTEGER)
)
SELECT 
    hour as "시간대",
    CASE 
        WHEN hour BETWEEN 7 AND 9 THEN '🌅 출근'
        WHEN hour BETWEEN 12 AND 14 THEN '🍽️ 점심'
        WHEN hour BETWEEN 18 AND 20 THEN '🌆 퇴근'
        WHEN hour BETWEEN 22 AND 23 THEN '🌙 심야'
        ELSE '⏰' 
    END as "구분",
    ROUND(hourly_revenue, 0) as "매출",
    unique_users as "순사용자",
    CASE WHEN hourly_impressions > 0 
         THEN ROUND(CAST(hourly_clicks AS DOUBLE) / hourly_impressions * 100, 2)
         ELSE 0 END as "CTR%",
    CASE WHEN hourly_clicks > 0 
         THEN ROUND(CAST(hourly_conversions AS DOUBLE) / hourly_clicks * 100, 2)
         ELSE 0 END as "CVR%",
    -- 피크타임 표시
    CASE 
        WHEN hourly_revenue = MAX(hourly_revenue) OVER () THEN '🔥 PEAK'
        WHEN hourly_revenue > AVG(hourly_revenue) OVER () * 1.2 THEN '📈 HIGH'
        WHEN hourly_revenue < AVG(hourly_revenue) OVER () * 0.8 THEN '📉 LOW'
        ELSE '➡️ NORMAL'
    END as "상태"
FROM hourly_today
ORDER BY hour
```

#### 3. 오늘의 TOP 5 캠페인 성과 (일별 집계 버전)
```sql
-- 오늘 ROAS 기준 상위 5개 캠페인 (conversion 포함)
-- ⚠️ 이 쿼리는 ad_combined_log_summary (일별) 사용 - 실시간성은 떨어지지만 ROAS 계산 가능
WITH today_campaigns AS (
    SELECT 
        campaign_id,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary  -- 일별 집계 테이블 유지 (conversion 데이터 필요)
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') = DATE '{{current_date}}'
    GROUP BY campaign_id
    HAVING SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0
)
SELECT 
    ROW_NUMBER() OVER (ORDER BY revenue/cost DESC) as "순위",
    campaign_id as "캠페인ID",
    ROUND(revenue, 0) as "매출",
    ROUND(cost, 0) as "비용",
    ROUND(revenue/cost, 2) as "ROAS",
    conversions as "전환수",
    CASE WHEN clicks > 0 
         THEN ROUND(CAST(conversions AS DOUBLE) / clicks * 100, 1)
         ELSE 0 END as "CVR%",
    -- 전체 매출 대비 기여도
    ROUND(100.0 * revenue / SUM(revenue) OVER (), 1) as "매출기여도%"
FROM today_campaigns
WHERE revenue > 0
ORDER BY revenue/cost DESC
LIMIT 5
```

#### 4. 실시간 이상 징후 알림
```sql
-- 최근 1시간 vs 평균 대비 이상 징후 감지
WITH recent_hour AS (
    SELECT 
        COUNT(*) as current_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as current_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as current_conversions,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as current_cost
    FROM ad_combined_log_summary
    WHERE 
        date_parse(CONCAT(
            CAST(year AS VARCHAR), '-', 
            LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
            LPAD(CAST(day AS VARCHAR), 2, '0'), ' ',
            LPAD(CAST(hour AS VARCHAR), 2, '0'), ':00:00'
        ), '%Y-%m-%d %H:%i:%s') >= CAST('{{current_date}}' AS TIMESTAMP) - INTERVAL '1' HOUR
),
avg_metrics AS (
    SELECT 
        AVG(hourly_impressions) as avg_impressions,
        AVG(hourly_clicks) as avg_clicks,
        AVG(hourly_conversions) as avg_conversions,
        AVG(hourly_cost) as avg_cost,
        STDDEV(hourly_impressions) as std_impressions,
        STDDEV(hourly_clicks) as std_clicks
    FROM (
        SELECT 
            date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
            CAST(hour AS INTEGER) as hour,
            COUNT(*) as hourly_impressions,
            SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as hourly_clicks,
            SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as hourly_conversions,
            SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as hourly_cost
        FROM ad_combined_log_summary
        WHERE 
            date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') >= DATE '{{current_date}}' - INTERVAL '7' DAY
            AND CAST(hour AS INTEGER) = HOUR(CAST('{{current_date}}' AS TIMESTAMP))
        GROUP BY 1, 2
    ) t
)
SELECT 
    DATE_FORMAT(CAST('{{current_date}}' AS TIMESTAMP), '%Y-%m-%d %H:%i') as "체크시각",
    CASE 
        WHEN current_impressions < avg_impressions - 2 * std_impressions THEN '⚠️ 노출수 급감'
        WHEN current_impressions > avg_impressions + 2 * std_impressions THEN '📈 노출수 급증'
        ELSE '✅ 정상'
    END as "노출수_상태",
    CASE 
        WHEN current_clicks < avg_clicks - 2 * std_clicks THEN '⚠️ 클릭수 급감'
        WHEN current_clicks > avg_clicks + 2 * std_clicks THEN '📈 클릭수 급증'
        ELSE '✅ 정상'
    END as "클릭수_상태",
    CASE 
        WHEN current_cost > avg_cost * 1.5 THEN '💸 비용 초과'
        WHEN current_cost < avg_cost * 0.5 THEN '💰 비용 절감'
        ELSE '✅ 정상'
    END as "비용_상태",
    current_impressions as "현재_노출수",
    ROUND(avg_impressions, 0) as "평균_노출수",
    current_clicks as "현재_클릭수",
    ROUND(avg_clicks, 0) as "평균_클릭수"
FROM recent_hour, avg_metrics
```

#### 5. 디바이스별 실시간 현황

-- 오늘 디바이스별 성과 분포
SELECT 
    device_type as "디바이스",
    COUNT(DISTINCT user_id) as "사용자수",
    ROUND(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END), 0) as "매출",

    SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",

    ROUND(100.0 * SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / 
          NULLIF(SUM(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END)) OVER (), 0), 1) as "매출_점유율%",
    -- CTR
    CASE WHEN COUNT(*) > 0 
         THEN ROUND(CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(*) * 100, 2)
         ELSE 0 END as "CTR%"
FROM ad_combined_log_summary
WHERE 
    date_parse(concat(CAST(year AS VARCHAR), '-', 
                     LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                     LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') = DATE '{{current_date}}'

ORDER BY "매출" DESC

#### 7. 주간 일별 트렌드 (이번주)
```sql
-- 최근한주 일별 성과 추이
WITH week_daily AS (
    SELECT 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
        day_of_week(date_parse(concat(CAST(year AS VARCHAR), '-', 
                                     LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                     LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as dow,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as daily_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as daily_cost,
        COUNT(DISTINCT user_id) as daily_users
    FROM ad_combined_log_summary
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        >= DATE '{{current_date}}' - INTERVAL '6' DAY 
        AND date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        <= DATE '{{current_date}}'
    GROUP BY 1, 2
)
SELECT 
    date_format(date, '%m-%d') as "날짜",
    CASE dow
        WHEN 1 THEN '일'
        WHEN 2 THEN '월'
        WHEN 3 THEN '화'
        WHEN 4 THEN '수'
        WHEN 5 THEN '목'
        WHEN 6 THEN '금'
        WHEN 7 THEN '토'
    END as "요일",
    ROUND(daily_revenue, 0) as "매출",
    ROUND(daily_cost, 0) as "비용",
    CASE WHEN daily_cost > 0 
         THEN ROUND(daily_revenue / daily_cost, 2)
         ELSE 0 END as "ROAS",
    daily_users as "순사용자",
    -- 주간 평균 대비
    ROUND(100.0 * daily_revenue / AVG(daily_revenue) OVER () - 100, 1) as "평균대비%"
FROM week_daily
ORDER BY date
```

#### 8. 주간 광고주별 ROAS TOP 5
```sql
-- 이번주 ROAS 상위 광고주
WITH weekly_advertisers AS (
    SELECT 
        advertiser_id,
        advertiser_name,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(DISTINCT campaign_id) as campaign_count,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN DATE '{{current_date}}' - INTERVAL '6' DAY AND DATE '{{current_date}}'
        AND (cost_per_impression + CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0
    GROUP BY advertiser_id, advertiser_name
    HAVING SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 1000000 -- 최소 100만원 이상 집행
)
SELECT 
    ROW_NUMBER() OVER (ORDER BY revenue/cost DESC) as "순위",
    CONCAT(SUBSTR(advertiser_id, 1, 4), '..', SUBSTR(advertiser_id, -4)) as "광고주ID",
    advertiser_name as "광고주명",
    ROUND(revenue, 0) as "매출",
    ROUND(cost, 0) as "광고비",
    ROUND(revenue/cost, 2) as "ROAS",
    campaign_count as "캠페인수",
    conversions as "전환수"
FROM weekly_advertisers
WHERE revenue > 0
ORDER BY revenue/cost DESC
LIMIT 5
```

### 📊 월간 지표 (2개) - 전략적 인사이트

#### 9. 월간 목표 대비 달성률
```sql
-- 이번달 목표 달성 현황 (current_date 파라미터 사용)
WITH monthly_actual AS (
    SELECT 
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as mtd_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as mtd_cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as mtd_conversions,
        COUNT(DISTINCT user_id) as mtd_users,
        COUNT(DISTINCT advertiser_id) as mtd_advertisers
    FROM ad_combined_log_summary
    WHERE 
        CAST(year AS INTEGER) = YEAR(DATE '{{current_date}}')
        AND CAST(month AS INTEGER) = MONTH(DATE '{{current_date}}')
        AND CAST(day AS INTEGER) <= DAY(DATE '{{current_date}}')
),
targets AS (
    SELECT 
        3000000000 as target_revenue,    -- 월 매출 목표 (30억)
        1500000000 as target_cost,       -- 월 광고비 예산 (15억)
        400000 as target_conversions,    -- 월 전환 목표 (40만건)
        DAY(DATE '{{current_date}}') as days_elapsed,
        -- 각 월의 일수를 직접 매핑
        CASE MONTH(DATE '{{current_date}}')
            WHEN 1 THEN 31
            WHEN 2 THEN CASE 
                WHEN MOD(YEAR(DATE '{{current_date}}'), 4) = 0 
                    AND (MOD(YEAR(DATE '{{current_date}}'), 100) != 0 
                    OR MOD(YEAR(DATE '{{current_date}}'), 400) = 0) 
                THEN 29 ELSE 28 END  -- 윤년 처리
            WHEN 3 THEN 31
            WHEN 4 THEN 30
            WHEN 5 THEN 31
            WHEN 6 THEN 30
            WHEN 7 THEN 31
            WHEN 8 THEN 31
            WHEN 9 THEN 30
            WHEN 10 THEN 31
            WHEN 11 THEN 30
            WHEN 12 THEN 31
        END as days_in_month
)
SELECT 
    DATE_FORMAT(DATE '{{current_date}}', '%Y-%m') as "월",
    CONCAT(days_elapsed, '/', days_in_month, '일') as "경과일",
    ROUND(days_elapsed * 100.0 / days_in_month, 1) as "월진행률%",
    -- 매출
    ROUND(mtd_revenue, 0) as "당월누적_매출",
    ROUND(target_revenue, 0) as "월목표_매출",
    ROUND(mtd_revenue / target_revenue * 100, 1) as "매출_달성률%",
    ROUND((mtd_revenue / days_elapsed) * days_in_month, 0) as "매출_예상착지",
    -- 비용
    ROUND(mtd_cost, 0) as "당월누적_비용",
    ROUND(target_cost, 0) as "월예산",
    ROUND(mtd_cost / target_cost * 100, 1) as "예산_소진율%",
    -- ROAS
    CASE WHEN mtd_cost > 0 
         THEN ROUND(mtd_revenue / mtd_cost, 2)
         ELSE 0 END as "누적_ROAS",
    -- 전환
    mtd_conversions as "당월_전환수",
    ROUND(mtd_conversions / target_conversions * 100, 1) as "전환_달성률%"
FROM monthly_actual
CROSS JOIN targets
```

#### 10. 당월 일별 매출 추이
```sql
-- 당월 일별 매출 및 ROAS 추이
WITH monthly_daily AS (
    SELECT 
        day,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as daily_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as daily_cost,
        COUNT(DISTINCT user_id) as daily_users,
        COUNT(DISTINCT advertiser_id) as daily_advertisers,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as daily_conversions
    FROM ad_combined_log_summary
    WHERE 
        CAST(year AS INTEGER) = YEAR(DATE '{{current_date}}')
        AND CAST(month AS INTEGER) = MONTH(DATE '{{current_date}}')
        AND CAST(day AS INTEGER) <= DAY(DATE '{{current_date}}')
    GROUP BY day
)
SELECT 
    day as "일자",
    CASE day_of_week(
            DATE_ADD('day', CAST(day AS INTEGER),
                DATE_TRUNC('month', DATE '{{current_date}}'))
        )
        WHEN 1 THEN '일요일'
        WHEN 2 THEN '월요일'
        WHEN 3 THEN '화요일'
        WHEN 4 THEN '수요일'
        WHEN 5 THEN '목요일'
        WHEN 6 THEN '금요일'
        WHEN 7 THEN '토요일'
    END as "요일",
    ROUND(daily_revenue, 0) as "일매출",
    -- 일평균 대비
    ROUND(100.0 * daily_revenue / AVG(daily_revenue) OVER () - 100, 1) as "일평균매출대비(%)",
    ROUND(daily_cost, 0) as "일비용",
    CASE WHEN daily_cost > 0 
         THEN ROUND(daily_revenue / daily_cost, 2)
         ELSE 0 END as "ROAS",
    daily_users as "순사용자",
    daily_conversions as "전환수",
    -- 누적 매출
    ROUND(SUM(daily_revenue) OVER (ORDER BY day), 0) as "누적_매출"
    FROM monthly_daily
ORDER BY day
```

## Redash 대시보드 구성 가이드

### 날짜 파라미터 상세 설정

```yaml
# Redash Query Parameter Configuration

# 1. 메인 날짜 파라미터 (필수)
current_date:
  type: Date
  default: "today"  # 기본값: 오늘 날짜
  title: "분석 기준 날짜"
  description: "모든 분석의 기준이 되는 날짜입니다"
  
# 파라미터 적용 범위:
# - 일간 지표: current_date 하루 데이터
# - 주간 지표: current_date 기준 최근 7일 (current_date - 6일 ~ current_date)
# - 월간 지표: current_date 기준 해당 월 1일부터 current_date까지

# 2. 추가 날짜 파라미터 (선택사항)
start_date:  # 범위 분석용 시작일
  type: Date
  default: "{{current_date}} - 7"  # 기본: 일주일 전
  title: "시작 날짜"
  
end_date:    # 범위 분석용 종료일
  type: Date
  default: "{{current_date}}"
  title: "종료 날짜"
```

### 자동 새로고침 설정

- **월간 지표 (9-10번)**: 6시간마다 자동 새로고침

### 파라미터 설정

```yaml
# Redash 대시보드 파라미터

# 1. 날짜 파라미터 (필수)
current_date: 
  type: Date
  default: "today"
  title: "분석 기준 날짜"
  description: "모든 분석의 기준이 되는 날짜"

# 2. 기간 파라미터 (선택사항 - 커스텀 분석용)
start_date:
  type: Date  
  default: "{{current_date}} - 7"
  title: "시작 날짜"
  description: "범위 조회 시 시작일"
  
end_date:
  type: Date
  default: "{{current_date}}"  
  title: "종료 날짜"
  description: "범위 조회 시 종료일"

# 파라미터 적용 규칙:
# - 일간 지표: current_date 기준 당일 데이터 표시
# - 주간 지표: current_date-6일부터 current_date까지 7일간 데이터
# - 월간 지표: current_date 기준 해당 월 1일부터 current_date까지 데이터
```

### 알림 설정 권장사항

1. **실시간 이상징후**: CTR 또는 비용이 평균 대비 ±50% 벗어날 때
2. **일일 목표**: 오후 6시 기준 일일 목표 달성률 80% 미만일 때
3. **주간 ROAS**: 주간 ROAS가 1.5 이하로 떨어질 때
4. **월간 예산**: 월 예산 소진율이 진행률 대비 120% 초과할 때

## 활용 시나리오

### 경영진용
- 매일 오전 9시: 전일 성과 및 당일 목표 확인
  - `current_date`를 어제 날짜로 설정하여 전일 성과 완전 분석
- 매주 월요일: 주간 성과 리뷰 및 전략 조정
  - `current_date`를 일요일로 설정하여 전주 성과 분석
- 매월 1일: 전월 성과 분석 및 당월 목표 설정
  - `current_date`를 전월 마지막 날로 설정하여 월간 성과 분석

### 운영팀용
- 실시간 모니터링: 이상징후 발생 시 즉시 대응
  - `current_date`를 오늘로 설정하고 매 5분마다 새로고침
- 피크타임 관리: 시간대별 광고 예산 조정
  - 특정 날짜의 시간대별 성과 분석
- 캠페인 최적화: 저성과 캠페인 조기 발견 및 개선

### 마케팅팀용
- 광고주 관리: TOP/Bottom 광고주 식별 및 관리
  - 특정 기간 선택하여 광고주별 성과 비교
- 디바이스 전략: 디바이스별 성과 기반 타겟팅 조정
  - `start_date`와 `end_date`로 원하는 기간 분석
- 성장 분석: 월간 트렌드 기반 중장기 전략 수립
