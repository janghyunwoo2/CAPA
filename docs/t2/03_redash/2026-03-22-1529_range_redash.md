# Redash 대시보드 구성 가이드

> **테이블 사용 가이드**
> - `ad_combined_log`: 시간별(hourly) 상세 데이터가 필요한 경우 (hour 파티션 포함)
> - `ad_combined_log_summary`: 일별(daily) 집계 데이터로 충분한 경우 (hour 파티션 없음)

## 1. 다중 기간 대시보드 (일간/주간/월간 구분)

### 1.1 일간 지표 (4개)
실시간 모니터링과 즉각적인 의사결정을 위한 지표

#### 1) 일별 핵심 성과 지표
```sql
-- 선택 기간 일별 성과 지표
-- 주의: ad_combined_log_summary 사용 (일별 집계 테이블)
SELECT 
    date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') as "날짜",
    SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
    COUNT(*) as "노출수",
    SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
    SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수",
    SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
    CASE WHEN COUNT(*) > 0 THEN ROUND(CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(*) * 100, 2) ELSE 0 END as "CTR",
    CASE WHEN SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0 
         THEN ROUND(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / 
                   (SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END)), 2)
         ELSE 0 END as "ROAS"
FROM ad_combined_log_summary
WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') 
      BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
GROUP BY date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')
ORDER BY "날짜" DESC
```

#### 2) 시간대별 평균 성과
```sql
-- 선택 기간 시간대별 평균 성과
-- 주의: ad_combined_log는 impression/click만 포함
SELECT 
    CAST(hour AS INTEGER) as "시간대",
    AVG(cost_per_impression + CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "평균_광고비",
    AVG(impression_count) as "평균_노출수",
    AVG(click_count) as "평균_클릭수",
    CASE WHEN SUM(impression_count) > 0 
         THEN ROUND(CAST(SUM(click_count) AS DOUBLE) / SUM(impression_count) * 100, 2) 
         ELSE 0 END as "평균_CTR"
FROM (
    SELECT 
        hour,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost_per_impression,
        COUNT(*) as impression_count,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as click_count,
        CASE WHEN is_click THEN cost_per_click ELSE 0 END as cost_per_click,
        is_click
    FROM ad_combined_log
    WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') 
          BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
    GROUP BY hour, date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d'), is_click, cost_per_click
) hourly_data
GROUP BY hour
ORDER BY CAST(hour AS INTEGER)
```

#### 3) 기간 내 Top 5 캠페인
```sql
-- 선택 기간 최고 성과 캠페인 TOP 5
SELECT 
    ROW_NUMBER() OVER (ORDER BY revenue DESC) as "순위",
    CONCAT(SUBSTR(campaign_id, 1, 4), '..', SUBSTR(campaign_id, -4)) as "캠페인ID",
    SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
    SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
    CASE 
        WHEN SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0 
        THEN ROUND(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / 
                  (SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END)), 2)
        ELSE 0 END as "ROAS",
    SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수"
FROM ad_combined_log_summary
WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') 
      BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
GROUP BY campaign_id
HAVING revenue > 0
ORDER BY revenue DESC
LIMIT 5
```

#### 4) 디바이스별 기간 성과
```sql
-- 선택 기간 디바이스별 성과 분포
SELECT 
    device_type as "디바이스",
    COUNT(DISTINCT user_id) as "사용자수",
    SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
    COUNT(*) as "노출수",
    SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
    CASE WHEN COUNT(*) > 0 
         THEN ROUND(CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(*) * 100, 2) 
         ELSE 0 END as "CTR",
    ROUND(100.0 * SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / 
          SUM(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END)) OVER (), 1) as "매출_점유율"
FROM ad_combined_log_summary
WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') 
      BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
GROUP BY device_type
ORDER BY "매출" DESC
```

### 1.2 주간 지표 (2개)
중기 트렌드 파악과 전략 조정을 위한 지표

#### 1) 주간 성과 트렌드
```sql
-- 선택 기간 일별 추이와 7일 이동평균
WITH daily_trend AS (
    SELECT 
        date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') as date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')
          BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
    GROUP BY 1
)
SELECT 
    date_format(date, '%Y-%m-%d') as "날짜",
    revenue as "매출",
    cost as "광고비",
    CASE WHEN cost > 0 THEN ROUND(revenue / cost, 2) ELSE 0 END as "ROAS",
    clicks as "클릭수",
    conversions as "전환수",
    AVG(revenue) OVER (ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) as "7일_이동평균_매출"
FROM daily_trend
ORDER BY date DESC
```

#### 2) 주간 시간대별 평균 성과
```sql
-- 선택 기간 시간대별 평균 성과 (피크타임 분석)
-- 주의: ad_combined_log는 impression/click만 포함
SELECT 
    CAST(hour AS INTEGER) as "시간대",
    CASE 
        WHEN CAST(hour AS INTEGER) BETWEEN 0 AND 5 THEN '새벽(0-6시)'
        WHEN CAST(hour AS INTEGER) BETWEEN 6 AND 11 THEN '오전(6-12시)'
        WHEN CAST(hour AS INTEGER) BETWEEN 12 AND 17 THEN '오후(12-18시)'
        WHEN CAST(hour AS INTEGER) BETWEEN 18 AND 23 THEN '저녁(18-24시)'
    END as "시간대_구분",
    AVG(cost_per_impression + CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "평균_시간당_광고비",
    COUNT(*) / COUNT(DISTINCT date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')) as "일평균_노출수",
    SUM(CASE WHEN is_click THEN 1 ELSE 0 END) / CAST(COUNT(DISTINCT date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')) AS DOUBLE) as "일평균_클릭수",
    CASE WHEN COUNT(*) > 0 
         THEN ROUND(CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(*) * 100, 2) 
         ELSE 0 END as "평균_CTR"
FROM ad_combined_log
WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                       lpad(CAST(month AS VARCHAR), 2, '0'), '-', 
                       lpad(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
      BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
GROUP BY hour
ORDER BY "평균_CTR" DESC
```

### 1.3 월간 지표 (2개)
장기 전략 수립과 목표 관리를 위한 지표

#### 1) 월간 목표 대비 달성률
```sql
-- 선택 기간 목표 대비 달성률
WITH period_performance AS (
    SELECT 
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as actual_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as actual_cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as actual_conversions,
        COUNT(DISTINCT advertiser_id) as active_advertisers,
        COUNT(DISTINCT campaign_id) as active_campaigns,
        MIN(date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')) as start_date,
        MAX(date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')) as end_date
    FROM ad_combined_log_summary
    WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')
          BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
)
SELECT 
    concat(date_format(start_date, '%Y-%m-%d'), ' ~ ', date_format(end_date, '%Y-%m-%d')) as "기간",
    CAST(date_diff('day', start_date, end_date) + 1 AS VARCHAR) || '일' as "기간일수",
    actual_revenue as "실적_매출",
    actual_cost as "실적_광고비",
    CASE WHEN actual_cost > 0 THEN ROUND(actual_revenue / actual_cost, 2) ELSE 0 END as "ROAS",
    actual_conversions as "총_전환수",
    active_advertisers as "활성_광고주수",
    active_campaigns as "활성_캠페인수",
    ROUND(actual_revenue / NULLIF(date_diff('day', start_date, end_date) + 1, 0), 0) as "일평균_매출",
    ROUND(actual_cost / NULLIF(date_diff('day', start_date, end_date) + 1, 0), 0) as "일평균_광고비"
FROM period_performance
```

#### 2) 월간 캠페인별 ROI 순위
```sql
-- 선택 기간 캠페인별 ROI TOP 10
WITH period_campaigns AS (
    SELECT 
        campaign_id,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(DISTINCT advertiser_id) as advertisers,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        lpad(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        lpad(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks
    FROM ad_combined_log_summary
    WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') 
          BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
    GROUP BY campaign_id
    HAVING cost > 0
)
SELECT 
    ROW_NUMBER() OVER (ORDER BY revenue / cost DESC) as "순위",
    CONCAT(SUBSTR(campaign_id, 1, 4), '..', SUBSTR(campaign_id, -4)) as "캠페인ID",
    revenue as "총_매출",
    cost as "총_비용",
    ROUND(revenue / cost, 2) as "ROI",
    ROUND((revenue - cost) / cost * 100, 1) as "수익률(%)",
    active_days as "활동일수",
    advertisers as "참여_광고주수",
    conversions as "전환수",
    CASE WHEN impressions > 0 THEN ROUND(CAST(clicks AS DOUBLE) / impressions * 100, 2) ELSE 0 END as "CTR"
FROM period_campaigns
ORDER BY revenue / cost DESC
LIMIT 10
```
