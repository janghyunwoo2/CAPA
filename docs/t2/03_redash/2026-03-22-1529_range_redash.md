# Redash 대시보드 구성 가이드

> **테이블 사용 가이드**
> - `ad_combined_log`: 시간별(hourly) 상세 데이터가 필요한 경우 (hour 파티션 포함)
> - `ad_combined_log_summary`: 일별(daily) 집계 데이터로 충분한 경우 (hour 파티션 없음)

## 1. 다중 기간 대시보드 (일간/주간/월간 구분)

### 1.1 일간 지표 (4개)
실시간 모니터링과 즉각적인 의사결정을 위한 지표

#### 1) 일별 핵심 성과 지표
```sql
-- 선택 기간 일별 성과 지표
-- 주의: ad_combined_log_summary 사용 (일별 집계 테이블)
SELECT 
    date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') as "날짜",
    ROUND(SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END), 0) as "광고비",
    COUNT(*) as "노출수",
    SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
    SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수",
    ROUND(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END), 0) as "매출",
    CASE WHEN COUNT(*) > 0 THEN ROUND(CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(*) * 100, 2) ELSE 0 END as "CTR",
    CASE WHEN SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0 
         THEN ROUND(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / 
                   (SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END)), 2)
         ELSE 0 END as "ROAS"
FROM ad_combined_log_summary
WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') 
      BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
GROUP BY date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')
ORDER BY "날짜" DESC
```

### 1.2 주간 지표 (2개)
중기 트렌드 파악과 전략 조정을 위한 지표

#### 1) 주간 성과 트렌드
```sql
-- 선택 기간 일별 추이와 7일 이동평균
WITH daily_trend AS (
    SELECT 
        date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d') as date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')
          BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
    GROUP BY 1
)
SELECT 
    date_format(date, '%Y-%m-%d') as "날짜",
    ROUND(revenue, 0) as "매출",
    ROUND(cost, 0) as "광고비",
    CASE WHEN cost > 0 THEN ROUND(revenue / cost, 2) ELSE 0 END as "ROAS",
    clicks as "클릭수",
    conversions as "전환수",
    ROUND(AVG(revenue) OVER (ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW), 0) as "7일_이동평균_매출"
FROM daily_trend
ORDER BY date DESC
```

### 1.3 월간 지표 (2개)
장기 전략 수립과 목표 관리를 위한 지표

#### 1) 월간 목표 대비 달성률
```sql
-- 선택 기간 목표 대비 달성률
WITH period_performance AS (
    SELECT 
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as actual_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as actual_cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as actual_conversions,
        COUNT(DISTINCT advertiser_id) as active_advertisers,
        COUNT(DISTINCT campaign_id) as active_campaigns,
        MIN(date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')) as start_date,
        MAX(date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')) as end_date
    FROM ad_combined_log_summary
    WHERE date_parse(concat(year, '-', month, '-', day), '%Y-%m-%d')
          BETWEEN date('{{ start_date }}') AND date('{{ end_date }}')
)
SELECT 
    concat(date_format(start_date, '%Y-%m-%d'), ' ~ ', date_format(end_date, '%Y-%m-%d')) as "기간",
    CAST(date_diff('day', start_date, end_date) + 1 AS VARCHAR) || '일' as "기간일수",
    ROUND(actual_revenue, 0) as "실적_매출",
    ROUND(actual_cost, 0) as "실적_광고비",
    CASE WHEN actual_cost > 0 THEN ROUND(actual_revenue / actual_cost, 2) ELSE 0 END as "ROAS",
    actual_conversions as "총_전환수",
    active_advertisers as "활성_광고주수",
    active_campaigns as "활성_캠페인수",
    ROUND(actual_revenue / NULLIF(date_diff('day', start_date, end_date) + 1, 0), 0) as "일평균_매출",
    ROUND(actual_cost / NULLIF(date_diff('day', start_date, end_date) + 1, 0), 0) as "일평균_광고비"
FROM period_performance
```
