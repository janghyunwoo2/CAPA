# 광고 분석 일간 쿼리 가이드

## 7. 실제 활용 예시 (Redash 날짜 파라미터 활용)

### 7.1 날짜 파라미터를 활용한 개선된 쿼리
Redash의 Date 파라미터를 사용하여 기본값(어제)과 수동 날짜 변경을 모두 지원합니다.

#### 일간 KPI 요약 (개선된 버전)
```sql
-- 전체 KPI 요약 - 기본값은 어제, 파라미터로 변경 가능
WITH date_params AS (
    SELECT 
        DATE('{{ target_date }}') as target_date,
        CAST(YEAR(DATE('{{ target_date }}')) AS VARCHAR) as year_val,
        LPAD(CAST(MONTH(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as month_val,
        LPAD(CAST(DAY(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as day_val
),
summary AS (
    SELECT 
        COUNT(*) as total_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as total_clicks,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as total_cost,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as total_revenue,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as total_conversions,
        COUNT(DISTINCT advertiser_id) as active_advertisers,
        COUNT(DISTINCT campaign_id) as active_campaigns,
        COUNT(DISTINCT ad_id) as active_ads
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        year = year_val
        AND month = month_val
        AND day = day_val
)
SELECT 
    target_date as "조회날짜",
    total_impressions as "총_노출수",
    total_clicks as "총_클릭수",
    total_cost as "총_광고비",
    total_revenue as "총_매출",
    total_conversions as "총_전환수",
    active_advertisers as "활성_광고주수",
    active_campaigns as "활성_캠페인수",
    active_ads as "활성_광고수",
    CASE WHEN total_impressions > 0 
         THEN ROUND(CAST(total_clicks AS DOUBLE) / total_impressions * 100, 2)
         ELSE 0 END as "전체_CTR",
    CASE WHEN total_clicks > 0 
         THEN ROUND(total_cost / total_clicks, 0)
         ELSE 0 END as "전체_CPC",
    CASE WHEN total_clicks > 0 
         THEN ROUND(CAST(total_conversions AS DOUBLE) / total_clicks * 100, 2)
         ELSE 0 END as "전체_CVR",
    CASE WHEN total_cost > 0 
         THEN ROUND(total_revenue / total_cost, 2)
         ELSE 0 END as "전체_ROAS"
FROM summary
CROSS JOIN date_params
```

#### 시간별 트렌드 분석 (개선된 버전)
```sql
-- 시간별 노출수와 클릭수 추이 - 기본값은 어제, 파라미터로 변경 가능
```

#### 시간대별 성과 비교
```sql
-- 시간대별 성과 비교 - 오전/오후/저녁/심야 구분
WITH date_params AS (
    SELECT 
        DATE('{{ target_date }}') as target_date,
        CAST(YEAR(DATE('{{ target_date }}')) AS VARCHAR) as year_val,
        LPAD(CAST(MONTH(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as month_val,
        LPAD(CAST(DAY(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as day_val
),
hourly_data AS (
    SELECT 
        hour,
        CASE 
            WHEN hour BETWEEN 0 AND 5 THEN '심야(00-06시)'
            WHEN hour BETWEEN 6 AND 11 THEN '오전(06-12시)'
            WHEN hour BETWEEN 12 AND 17 THEN '오후(12-18시)'
            ELSE '저녁(18-24시)'
        END as time_zone,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log
    CROSS JOIN date_params
    WHERE 
        year = year_val
        AND month = month_val
        AND day = day_val
    GROUP BY hour
)
SELECT 
    target_date as "조회날짜",
    time_zone as "시간대",
    SUM(impressions) as "노출수",
    SUM(clicks) as "클릭수",
    SUM(cost) as "광고비",
    SUM(revenue) as "매출",
    SUM(conversions) as "전환수",
    ROUND(SUM(impressions) * 100.0 / SUM(SUM(impressions)) OVER(), 1) as "노출_비중%",
    CASE WHEN SUM(impressions) > 0 
         THEN ROUND(CAST(SUM(clicks) AS DOUBLE) / SUM(impressions) * 100, 2)
         ELSE 0 END as "CTR",
    CASE WHEN SUM(cost) > 0 
         THEN ROUND(SUM(revenue) / SUM(cost), 2)
         ELSE 0 END as "ROAS"
FROM hourly_data
CROSS JOIN date_params
GROUP BY target_date, time_zone
ORDER BY 
    CASE time_zone
        WHEN '심야(00-06시)' THEN 1
        WHEN '오전(06-12시)' THEN 2
        WHEN '오후(12-18시)' THEN 3
        WHEN '저녁(18-24시)' THEN 4
    END
```

#### 일간 목표 달성률 분석
```sql
-- 일간 목표 대비 실적 달성률 - 파라미터로 목표값 조정 가능
WITH date_params AS (
    SELECT 
        DATE('{{ target_date }}') as target_date,
        CAST(YEAR(DATE('{{ target_date }}')) AS VARCHAR) as year_val,
        LPAD(CAST(MONTH(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as month_val,
        LPAD(CAST(DAY(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as day_val
),
daily_actual AS (
    SELECT 
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as actual_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as actual_cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as actual_conversions,
        COUNT(*) as actual_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as actual_clicks
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        year = year_val
        AND month = month_val
        AND day = day_val
),
targets AS (
    SELECT 
        {{ daily_revenue_target|100000000 }} as target_revenue,  -- 기본 1억
        {{ daily_cost_target|50000000 }} as target_cost,         -- 기본 5천만
        {{ daily_conversion_target|10000 }} as target_conversions  -- 기본 1만
)
SELECT 
    target_date as "조회날짜",
    actual_revenue as "실적_매출",
    target_revenue as "목표_매출",
    ROUND(actual_revenue / target_revenue * 100, 1) as "매출_달성률%",
    actual_cost as "실적_광고비",
    target_cost as "목표_광고비",
    ROUND(actual_cost / target_cost * 100, 1) as "광고비_집행률%",
    actual_conversions as "실적_전환수",
    target_conversions as "목표_전환수",
    ROUND(actual_conversions / target_conversions * 100, 1) as "전환_달성률%",
    CASE WHEN actual_cost > 0 
         THEN ROUND(actual_revenue / actual_cost, 2)
         ELSE 0 END as "ROAS",
    actual_impressions as "노출수",
    actual_clicks as "클릭수"
FROM daily_actual
CROSS JOIN targets
CROSS JOIN date_params
```

### 7.2 Redash 파라미터 설정 방법
1. 쿼리 편집 화면에서 `{{ target_date }}` 파라미터 자동 생성됨
2. 파라미터 타입을 "Date"로 설정
3. 기본값: "yesterday" 선택 (어제 날짜 자동 설정)

### 7.3 장점
- **자동화**: 대시보드가 매일 자동으로 전일 데이터 조회
- **유연성**: 필요시 특정 날짜 선택하여 과거 데이터 분석 가능
- **단일 쿼리**: 파라미터와 CURRENT_DATE 방식을 통합하여 관리 편의성 향상
- **명확성**: 조회날짜가 결과에 표시되어 데이터 기준일 명확

### 7.4 추가 실무 활용 예시

#### 전일 대비 변화율 분석
```sql
-- 오늘과 어제의 주요 지표 비교 및 변화율 계산
WITH dates AS (
    SELECT 
        DATE('{{ target_date }}') as today,
        DATE('{{ target_date }}') - INTERVAL '1' day as yesterday
),
today_data AS (
    SELECT 
        'today' as day_type,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN dates
    WHERE 
        year = CAST(YEAR(today) AS VARCHAR)
        AND month = LPAD(CAST(MONTH(today) AS VARCHAR), 2, '0')
        AND day = LPAD(CAST(DAY(today) AS VARCHAR), 2, '0')
),
yesterday_data AS (
    SELECT 
        'yesterday' as day_type,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN dates
    WHERE 
        year = CAST(YEAR(yesterday) AS VARCHAR)
        AND month = LPAD(CAST(MONTH(yesterday) AS VARCHAR), 2, '0')
        AND day = LPAD(CAST(DAY(yesterday) AS VARCHAR), 2, '0')
```

#### 비용 효율성 분석 (CPC, CPM, CPA)
```sql
-- 일간 비용 효율성 지표 분석 - CPC, CPM, CPA 추이
WITH date_params AS (
    SELECT 
        DATE('{{ target_date }}') as target_date,
        CAST(YEAR(DATE('{{ target_date }}')) AS VARCHAR) as year_val,
        LPAD(CAST(MONTH(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as month_val,
        LPAD(CAST(DAY(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as day_val
),
hourly_metrics AS (
    SELECT 
        hour,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log
    CROSS JOIN date_params
    WHERE 
        year = year_val
        AND month = month_val
        AND day = day_val
    GROUP BY hour
)
SELECT 
    target_date as "조회날짜",
    CAST(hour AS INTEGER) as "시간",
    impressions as "노출수",
    clicks as "클릭수",
    conversions as "전환수",
    cost as "광고비",
    CASE WHEN impressions >= 1000 
         THEN ROUND(cost / (impressions / 1000.0), 0)
         ELSE 0 END as "CPM",
    CASE WHEN clicks > 0 
         THEN ROUND(cost / clicks, 0)
         ELSE 0 END as "CPC",
    CASE WHEN conversions > 0 
         THEN ROUND(cost / conversions, 0)
         ELSE 0 END as "CPA"
FROM hourly_metrics
CROSS JOIN date_params
ORDER BY CAST(hour AS INTEGER)
```

#### 키워드별 성과 분석
```sql
-- 키워드별 일간 성과 TOP 30 - 가장 효과적인 키워드 파악
WITH date_params AS (
    SELECT 
        DATE('{{ target_date }}') as target_date,
        CAST(YEAR(DATE('{{ target_date }}')) AS VARCHAR) as year_val,
        LPAD(CAST(MONTH(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as month_val,
        LPAD(CAST(DAY(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as day_val
)
SELECT 
    target_date as "조회날짜",
    keyword as "키워드",
    COUNT(*) as "노출수",
    SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
    SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
    SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
    SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수",
    CASE WHEN COUNT(*) > 0 
         THEN ROUND(CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(*) * 100, 2)
         ELSE 0 END as "CTR",
    CASE WHEN SUM(CASE WHEN is_click THEN 1 ELSE 0 END) > 0 
         THEN ROUND(SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) / 
                    SUM(CASE WHEN is_click THEN 1 ELSE 0 END), 0)
         ELSE 0 END as "CPC",
    CASE WHEN SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0 
         THEN ROUND(SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / 
                   (SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END)), 2)
         ELSE 0 END as "ROAS"
FROM ad_combined_log_summary
CROSS JOIN date_params
WHERE 
    year = year_val
    AND month = month_val
    AND day = day_val
GROUP BY target_date, keyword
HAVING COUNT(*) >= 50  -- 최소 노출수 조건
ORDER BY "매출" DESC
LIMIT 30
```

#### 저성과 광고 리스트 (개선 필요)
```sql
-- 저성과 광고 리스트 - CTR이 낮거나 ROAS가 나쁜 광고 파악
WITH date_params AS (
    SELECT 
        DATE('{{ target_date }}') as target_date,
        CAST(YEAR(DATE('{{ target_date }}')) AS VARCHAR) as year_val,
        LPAD(CAST(MONTH(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as month_val,
        LPAD(CAST(DAY(DATE('{{ target_date }}')) AS VARCHAR), 2, '0') as day_val
),
ad_performance AS (
    SELECT 
        ad_id,
        campaign_id,
        advertiser_id,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        year = year_val
        AND month = month_val
        AND day = day_val
    GROUP BY ad_id, campaign_id, advertiser_id
),
avg_metrics AS (
    SELECT 
        AVG(CASE WHEN impressions > 0 THEN clicks * 100.0 / impressions ELSE 0 END) as avg_ctr,
        AVG(CASE WHEN cost > 0 THEN revenue / cost ELSE 0 END) as avg_roas
    FROM ad_performance
    WHERE impressions >= 100  -- 의미있는 노출수만 포함
)
SELECT 
    target_date as "조회날짜",
    ap.ad_id as "광고ID",
    ap.campaign_id as "캠페인ID",
    ap.advertiser_id as "광고주ID",
    ap.impressions as "노출수",
    ap.clicks as "클릭수",
    ap.cost as "광고비",
    ap.revenue as "매출",
    ap.conversions as "전환수",
    CASE WHEN ap.impressions > 0 
         THEN ROUND(ap.clicks * 100.0 / ap.impressions, 2)
         ELSE 0 END as "CTR",
    am.avg_ctr as "평균_CTR",
    CASE WHEN ap.cost > 0 
         THEN ROUND(ap.revenue / ap.cost, 2)
         ELSE 0 END as "ROAS",
    am.avg_roas as "평균_ROAS",
    CASE 
        WHEN ap.impressions > 0 AND ap.clicks * 100.0 / ap.impressions < am.avg_ctr * 0.5 
        THEN 'CTR 매우 낮음'
        WHEN ap.cost > 0 AND ap.revenue / ap.cost < am.avg_roas * 0.5 
        THEN 'ROAS 매우 낮음'
        WHEN ap.impressions > 1000 AND ap.clicks = 0 
        THEN '클릭 전무'
        WHEN ap.cost > 100000 AND ap.revenue = 0 
        THEN '전환 전무'
        ELSE '기타'
    END as "문제점"
FROM ad_performance ap
CROSS JOIN avg_metrics am
CROSS JOIN date_params
WHERE 
    (ap.impressions >= 100 AND ap.clicks * 100.0 / ap.impressions < am.avg_ctr * 0.5)
    OR (ap.cost >= 10000 AND ap.revenue / ap.cost < am.avg_roas * 0.5)
    OR (ap.impressions > 1000 AND ap.clicks = 0)
    OR (ap.cost > 100000 AND ap.revenue = 0)
ORDER BY ap.cost DESC
LIMIT 50
```
