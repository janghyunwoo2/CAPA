## 2. 통합 기간 파라미터 대시보드 (기간 구분 없음)

기간을 자유롭게 설정할 수 있는 범용 대시보드 (start_date, end_date 파라미터 사용)

### 1) 기간 종합 성과 요약
```sql
-- 기간 종합 성과 한눈에 보기
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as days
),
period_summary AS (
    SELECT 
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions,
        COUNT(DISTINCT user_id) as unique_users,
        COUNT(DISTINCT advertiser_id) as advertisers,
        COUNT(DISTINCT campaign_id) as campaigns
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN start_date AND end_date
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    days as "기간(일)",
    revenue as "총매출",
    cost as "총광고비",
    CASE WHEN cost > 0 THEN ROUND(revenue / cost, 2) ELSE 0 END as "ROAS",
    impressions as "총노출수",
    clicks as "총클릭수",
    conversions as "총전환수",
    unique_users as "순사용자수",
    advertisers as "광고주수",
    campaigns as "캠페인수",
    ROUND(revenue / days, 0) as "일평균_매출",
    ROUND(cost / days, 0) as "일평균_광고비"
FROM period_summary
CROSS JOIN date_params
```

### 2) 기간 일별 추이

```sql
-- 기간 내 일별 성과 추이
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
daily_performance AS (
    SELECT 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN start_date AND end_date
    GROUP BY 1
)
SELECT 
    date_format(date, '%Y-%m-%d') as "날짜",
    revenue as "매출",
    cost as "광고비",
    CASE WHEN cost > 0 THEN ROUND(revenue / cost, 2) ELSE 0 END as "ROAS",
    conversions as "전환수",
    AVG(revenue) OVER (ORDER BY date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) as "7일_이동평균_매출"
FROM daily_performance
ORDER BY date
```

### 4) 기간 캠페인별 ROI
```sql
-- 기간 내 캠페인별 ROI TOP 10
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
campaign_roi AS (
    SELECT 
        campaign_id,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN start_date AND end_date
      AND cost_per_impression + CASE WHEN is_click THEN cost_per_click ELSE 0 END > 0
    GROUP BY campaign_id
    HAVING SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0
)
SELECT 
    ROW_NUMBER() OVER (ORDER BY revenue / cost DESC) as "순위",
    CONCAT(SUBSTR(campaign_id, 1, 4), '..', SUBSTR(campaign_id, -4)) as "캠페인ID",
    revenue as "매출",
    cost as "광고비",
    ROUND(revenue / cost, 2) as "ROI",
    conversions as "전환수",
    CASE WHEN impressions > 0 THEN ROUND(CAST(clicks AS DOUBLE) / impressions * 100, 2) ELSE 0 END as "CTR",
    CASE WHEN clicks > 0 THEN ROUND(CAST(conversions AS DOUBLE) / clicks * 100, 2) ELSE 0 END as "CVR"
FROM campaign_roi
ORDER BY revenue / cost DESC
LIMIT 10
```

### 7) 기간 전환 퍼널
```sql
-- 기간 내 전환 퍼널 분석
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
funnel_data AS (
    SELECT 
        COUNT(*) as impressions,
        COUNT(DISTINCT user_id) as unique_users,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        COUNT(DISTINCT CASE WHEN is_click THEN user_id END) as users_clicked,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions,
        COUNT(DISTINCT CASE WHEN is_conversion THEN user_id END) as users_converted,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN start_date AND end_date
)
SELECT 
    '노출' as "단계",
    impressions as "이벤트수",
    unique_users as "사용자수",
    100.0 as "전체대비%",
    100.0 as "이전단계대비%"
FROM funnel_data
UNION ALL
SELECT 
    '클릭',
    clicks,
    users_clicked,
    ROUND(100.0 * clicks / impressions, 2),
    ROUND(100.0 * clicks / impressions, 2)
FROM funnel_data
UNION ALL
SELECT 
    '전환',
    conversions,
    users_converted,
    ROUND(100.0 * conversions / impressions, 2),
    ROUND(100.0 * conversions / clicks, 2)
FROM funnel_data
```

### 9) 기간 비교 분석
```sql
-- 두 기간 비교 (현재 vs 이전)
WITH current_period AS (
    SELECT 
        '현재기간' as period,
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN DATE('{{ start_date }}') AND DATE('{{ end_date }}')
),
previous_period AS (
    SELECT 
        '이전기간' as period,
        DATE_SUB(DATE('{{ start_date }}'), INTERVAL DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 DAY) as start_date,
        DATE_SUB(DATE('{{ start_date }}'), INTERVAL 1 DAY) as end_date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN DATE_SUB(DATE('{{ start_date }}'), INTERVAL DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 DAY)
              AND DATE_SUB(DATE('{{ start_date }}'), INTERVAL 1 DAY)
)
SELECT 
    period as "구분",
    date_format(start_date, '%m/%d') || '~' || date_format(end_date, '%m/%d') as "기간",
    revenue as "매출",
    cost as "광고비",
    CASE WHEN cost > 0 THEN ROUND(revenue / cost, 2) ELSE 0 END as "ROAS",
    impressions as "노출수",
    clicks as "클릭수",
    conversions as "전환수"
FROM (
    SELECT * FROM current_period
    UNION ALL
    SELECT * FROM previous_period
)
```

### 10) 기간 신규 광고주 성과
```sql
-- 기간 내 신규 광고주 성과 분석
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
before_period AS (
    SELECT DISTINCT advertiser_id
    FROM ad_combined_log_summary
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          < (SELECT start_date FROM date_params)
),
new_advertisers AS (
    SELECT 
        ads.advertiser_id,
        MIN(date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as first_date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(DISTINCT campaign_id) as campaigns
    FROM ad_combined_log_summary ads
    CROSS JOIN date_params
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN start_date AND end_date
      AND ads.advertiser_id NOT IN (SELECT advertiser_id FROM before_period)
    GROUP BY ads.advertiser_id
)
SELECT 
    CONCAT(SUBSTR(advertiser_id, 1, 4), '..', SUBSTR(advertiser_id, -4)) as "광고주ID",
    date_format(first_date, '%Y-%m-%d') as "첫광고일",
    revenue as "매출",
    cost as "광고비",
    CASE WHEN cost > 0 THEN ROUND(revenue / cost, 2) ELSE 0 END as "ROAS",
    campaigns as "캠페인수"
FROM new_advertisers
WHERE revenue > 0
ORDER BY revenue DESC
```

## 대시보드 구성 가이드

### 다중 기간 대시보드 레이아웃
1. **상단**: 일간 실시간 지표 (5개)
   - 자동 새로고침 설정 (5분 간격)
   - 시각적 알림 (예산 초과, 성과 급변동)

2. **중단**: 주간 트렌드 지표 (3개)
   - 차트 중심 시각화
   - 주간 비교 강조

3. **하단**: 월간 전략 지표 (2개)
   - 목표 대비 진행률 게이지
   - 장기 ROI 분석

### 통합 기간 대시보드 레이아웃
1. **파라미터 영역**: start_date, end_date 선택
2. **핵심 요약**: 종합 성과 카드 (지표 1)
3. **시계열 분석**: 일별 추이 차트 (지표 2)
4. **순위 분석**: 광고주, 캠페인 TOP 10 (지표 3, 4)
5. **세부 분석**: 시간대, 디바이스, 퍼널 등 (지표 5-10)

### 추천 시각화
- **카드**: 핵심 KPI (매출, ROAS, 전환수)
- **라인 차트**: 시계열 추이
- **바 차트**: 순위, 비교 분석
- **파이 차트**: 점유율, 분포
- **퍼널 차트**: 전환 단계
- **게이지**: 목표 달성률
