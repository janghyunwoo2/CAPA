# 광고 분석 주간 쿼리 가이드

## 7. 실제 활용 예시 (Redash 날짜 파라미터 활용)

### 7.1 주간 분석 개선된 쿼리

#### 주간 범위를 파라미터로 받는 방식
```sql
-- 주간 광고주별 성과 - 날짜 파라미터로 주간 선택 가능
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end
),
weekly_summary AS (
    SELECT 
        advertiser_id,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as weekly_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as weekly_cost,
        COUNT(*) as weekly_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as weekly_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as weekly_conversions
    FROM ad_combined_log_summary
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
    GROUP BY advertiser_id
)
SELECT 
    week_start as "주_시작일",
    week_end as "주_종료일",
    CONCAT(SUBSTR(advertiser_id, 1, 4), '..', SUBSTR(advertiser_id, -4)) as "광고주ID",
    weekly_revenue as "주간_매출",
    weekly_cost as "주간_광고비",
    weekly_impressions as "주간_노출수",
    weekly_clicks as "주간_클릭수",
    weekly_conversions as "주간_전환수",
    CASE WHEN weekly_cost > 0 
         THEN ROUND(weekly_revenue / weekly_cost, 2)
         ELSE 0 END as "주간_ROAS"
FROM weekly_summary
CROSS JOIN week_params
ORDER BY weekly_revenue DESC
```

#### 자동 이번주/지난주 비교 (개선된 버전)
```sql
-- 이번주 vs 지난주 비교 - 기준 날짜 파라미터 사용
WITH base_date AS (
    SELECT DATE('{{ reference_date }}') as ref_date
),
week_ranges AS (
    SELECT 
        date_trunc('week', ref_date) as this_week_start,
        ref_date as this_week_end,
        date_trunc('week', ref_date - interval '7' day) as last_week_start,
        date_trunc('week', ref_date) - interval '1' day as last_week_end
    FROM base_date
),
this_week AS (
    SELECT 
        '이번주' as "기간",
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
        COUNT(*) as "노출수",
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수"
    FROM ad_combined_log_summary
    CROSS JOIN week_ranges
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN this_week_start AND this_week_end
),
last_week AS (
    SELECT 
        '지난주' as "기간",
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
        COUNT(*) as "노출수",
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수"
    FROM ad_combined_log_summary
    CROSS JOIN week_ranges
    WHERE date_parse(concat(CAST(year AS VARCHAR), '-', 
                           LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                           LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
          BETWEEN last_week_start AND last_week_end
)
SELECT 
    ref_date as "기준날짜",
    "기간",
    "매출",
    "광고비",
    "노출수",
    "클릭수",
    "전환수",
    CASE WHEN "광고비" > 0 THEN ROUND("매출" / "광고비", 2) ELSE 0 END as "ROAS",
    CASE WHEN "노출수" > 0 
         THEN ROUND(CAST("클릭수" AS DOUBLE) / "노출수" * 100, 2) 
         ELSE 0 END as "CTR"
FROM (
    SELECT * FROM this_week
    UNION ALL
    SELECT * FROM last_week
)
CROSS JOIN base_date
```

### 7.2 Redash 설정
- **reference_date**: Date 타입, 기본값 "today" (오늘 기준)
- **week_end_date**: Date 타입, 주간 종료일 선택 (시작일은 자동으로 6일 전으로 계산)

#### 주간 일별 성과 추이
```sql
-- 주간 일별 성과 추이 - 날짜 파라미터 사용
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end
),
daily_performance AS (
    SELECT 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
        day_of_week(date_parse(concat(CAST(year AS VARCHAR), '-', 
                               LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                               LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as dow,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as daily_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as daily_cost,
        COUNT(*) as daily_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as daily_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as daily_conversions
    FROM ad_combined_log_summary
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
    GROUP BY 1, 2
)
SELECT 
    date_format(week_start, '%Y-%m-%d') || ' ~ ' || date_format(week_end, '%Y-%m-%d') as "주간",
    date_format(date, '%Y-%m-%d') as "날짜",
    CASE dow
        WHEN 1 THEN '일요일'
        WHEN 2 THEN '월요일'
        WHEN 3 THEN '화요일'
        WHEN 4 THEN '수요일'
        WHEN 5 THEN '목요일'
        WHEN 6 THEN '금요일'
        WHEN 7 THEN '토요일'
    END as "요일",
    daily_revenue as "일별_매출",
    daily_cost as "일별_광고비",
    daily_impressions as "일별_노출수",
    daily_clicks as "일별_클릭수",
    daily_conversions as "일별_전환수",
    CASE WHEN daily_cost > 0 
         THEN ROUND(daily_revenue / daily_cost, 2)
         ELSE 0 END as "일별_ROAS",
    CASE WHEN daily_impressions > 0 
         THEN ROUND(CAST(daily_clicks AS DOUBLE) / daily_impressions * 100, 2) 
         ELSE 0 END as "일별_CTR"
FROM daily_performance dp
CROSS JOIN week_params wp
ORDER BY date
```

#### 주간 캠페인별 ROAS 순위
```sql
-- 주간 캠페인별 ROAS TOP 10 - 날짜 파라미터 사용
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end
),
campaign_weekly AS (
    SELECT 
        campaign_id,
        campaign_name,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
        AND cost_per_impression + CASE WHEN is_click THEN cost_per_click ELSE 0 END > 0
    GROUP BY campaign_id, campaign_name
    HAVING SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) > 0
)
SELECT 
    date_format(week_start, '%Y-%m-%d') || ' ~ ' || date_format(week_end, '%Y-%m-%d') as "주간",
    ROW_NUMBER() OVER (ORDER BY revenue / cost DESC) as "순위",
    CONCAT(SUBSTR(campaign_id, 1, 4), '..', SUBSTR(campaign_id, -4)) as "캠페인ID",
    campaign_name as "캠페인명",
    revenue as "매출",
    cost as "광고비",
    ROUND(revenue / cost, 2) as "ROAS",
    impressions as "노출수",
    clicks as "클릭수",
    conversions as "전환수",
    CASE WHEN impressions > 0 
         THEN ROUND(CAST(clicks AS DOUBLE) / impressions * 100, 2) 
         ELSE 0 END as "CTR",
    CASE WHEN clicks > 0 
         THEN ROUND(CAST(conversions AS DOUBLE) / clicks * 100, 2) 
         ELSE 0 END as "CVR"
FROM campaign_weekly cw
CROSS JOIN week_params wp
ORDER BY revenue / cost DESC
LIMIT 10
```

#### 주간 신규 광고주 성과
```sql
-- 주간 신규 광고주 성과 분석 - 날짜 파라미터 사용
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end
),
first_week_advertisers AS (
    SELECT DISTINCT advertiser_id
    FROM ad_combined_log_summary
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        < (SELECT week_start FROM week_params)
),
new_advertisers AS (
    SELECT 
        ads.advertiser_id,
        ads.advertiser_name,
        MIN(date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as first_ad_date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as cost,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary ads
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
        AND ads.advertiser_id NOT IN (SELECT advertiser_id FROM first_week_advertisers)
    GROUP BY ads.advertiser_id, ads.advertiser_name
)
SELECT 
    date_format(week_start, '%Y-%m-%d') || ' ~ ' || date_format(week_end, '%Y-%m-%d') as "주간",
    CONCAT(SUBSTR(advertiser_id, 1, 4), '..', SUBSTR(advertiser_id, -4)) as "광고주ID",
    advertiser_name as "광고주명",
    date_format(first_ad_date, '%Y-%m-%d') as "첫광고일",
    revenue as "첫주_매출",
    cost as "첫주_광고비",
    impressions as "첫주_노출수",
    clicks as "첫주_클릭수",
    conversions as "첫주_전환수",
    CASE WHEN cost > 0 
         THEN ROUND(revenue / cost, 2)
         ELSE 0 END as "첫주_ROAS"
FROM new_advertisers na
CROSS JOIN week_params wp
ORDER BY revenue DESC
```

#### 주간 피크타임 분석
```sql
-- 주간 시간대별 피크타임 분석 - 날짜 파라미터 사용
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end
),
hourly_stats AS (
    SELECT 
        hour,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as days_active,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as hourly_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as hourly_cost,
        COUNT(*) as hourly_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as hourly_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as hourly_conversions,
        COUNT(DISTINCT user_id) as unique_users
    FROM ad_combined_log_summary
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
    GROUP BY hour
)
SELECT 
    date_format(week_start, '%Y-%m-%d') || ' ~ ' || date_format(week_end, '%Y-%m-%d') as "주간",
    hour as "시간대",
    CASE 
        WHEN hour BETWEEN 0 AND 5 THEN '새벽(0-6시)'
        WHEN hour BETWEEN 6 AND 11 THEN '오전(6-12시)'
        WHEN hour BETWEEN 12 AND 17 THEN '오후(12-18시)'
        WHEN hour BETWEEN 18 AND 23 THEN '저녁(18-24시)'
    END as "시간대_구분",
    days_active as "활동일수",
    hourly_revenue as "시간대_매출",
    hourly_cost as "시간대_광고비",
    hourly_impressions as "시간대_노출수",
    hourly_clicks as "시간대_클릭수",
    hourly_conversions as "시간대_전환수",
    unique_users as "순사용자수",
    ROUND(hourly_revenue / days_active, 0) as "일평균_매출",
    CASE WHEN hourly_cost > 0 
         THEN ROUND(hourly_revenue / hourly_cost, 2)
         ELSE 0 END as "시간대_ROAS",
    CASE WHEN hourly_impressions > 0 
         THEN ROUND(CAST(hourly_clicks AS DOUBLE) / hourly_impressions * 100, 2) 
         ELSE 0 END as "시간대_CTR"
FROM hourly_stats hs
CROSS JOIN week_params wp
ORDER BY hourly_revenue DESC
```

#### 주간 광고 피로도 분석
```sql
-- 주간 사용자 광고 노출 빈도 및 피로도 분석 - 날짜 파라미터 사용  
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end
),
user_exposure AS (
    SELECT 
        user_id,
        COUNT(*) as ad_exposures,
        COUNT(DISTINCT advertiser_id) as unique_advertisers,
        COUNT(DISTINCT campaign_id) as unique_campaigns,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as total_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as total_conversions,
        MIN(date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as first_exposure,
        MAX(date_parse(concat(CAST(year AS VARCHAR), '-', 
                             LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                             LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as last_exposure
    FROM ad_combined_log_summary
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
    GROUP BY user_id
),
exposure_buckets AS (
    SELECT 
        CASE 
            WHEN ad_exposures <= 10 THEN '01. 1-10회'
            WHEN ad_exposures <= 30 THEN '02. 11-30회'
            WHEN ad_exposures <= 50 THEN '03. 31-50회'
            WHEN ad_exposures <= 100 THEN '04. 51-100회'
            ELSE '05. 100회 초과'
        END as exposure_bucket,
        COUNT(*) as user_count,
        SUM(ad_exposures) as total_exposures,
        SUM(total_clicks) as total_clicks,
        SUM(total_conversions) as total_conversions
    FROM user_exposure
    GROUP BY 1
)
SELECT 
    date_format(week_start, '%Y-%m-%d') || ' ~ ' || date_format(week_end, '%Y-%m-%d') as "주간",
    exposure_bucket as "노출빈도_구간",
    user_count as "사용자수",
    total_exposures as "총노출수",
    total_clicks as "총클릭수",
    total_conversions as "총전환수",
    ROUND(CAST(total_exposures AS DOUBLE) / user_count, 1) as "사용자당_평균노출",
    CASE WHEN total_exposures > 0 
         THEN ROUND(CAST(total_clicks AS DOUBLE) / total_exposures * 100, 2) 
         ELSE 0 END as "CTR",
    CASE WHEN total_clicks > 0 
         THEN ROUND(CAST(total_conversions AS DOUBLE) / total_clicks * 100, 2) 
         ELSE 0 END as "CVR",
    ROUND(100.0 * user_count / SUM(user_count) OVER (), 1) as "사용자_비율"
FROM exposure_buckets eb
CROSS JOIN week_params wp
ORDER BY exposure_bucket
```

#### 주간 예산 소진율 분석
```sql
-- 주간 광고주별 예산 소진율 분석 - 날짜 파라미터 사용
WITH week_params AS (
    SELECT 
        DATE('{{ week_end_date }}') - INTERVAL '6' DAY as week_start,
        DATE('{{ week_end_date }}') as week_end,
        7 as week_days
),
advertiser_budget AS (
    SELECT 
        advertiser_id,
        advertiser_name,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as spent_budget,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions
    FROM ad_combined_log_summary
    CROSS JOIN week_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN week_start AND week_end
    GROUP BY advertiser_id, advertiser_name
)
SELECT 
    date_format(week_start, '%Y-%m-%d') || ' ~ ' || date_format(week_end, '%Y-%m-%d') as "주간",
    CONCAT(SUBSTR(advertiser_id, 1, 4), '..', SUBSTR(advertiser_id, -4)) as "광고주ID",
    advertiser_name as "광고주명",
    spent_budget as "소진_예산",
    {{ weekly_budget_limit|5000000 }} as "주간_예산한도",
    ROUND(spent_budget / {{ weekly_budget_limit|5000000 }} * 100, 1) as "예산_소진율",
    active_days || '/' || week_days as "활동일수",
    ROUND(spent_budget / active_days, 0) as "일평균_소진액",
    revenue as "매출",
    CASE WHEN spent_budget > 0 
         THEN ROUND(revenue / spent_budget, 2)
         ELSE 0 END as "ROAS",
    CASE 
        WHEN spent_budget / {{ weekly_budget_limit|5000000 }} >= 0.95 THEN '예산소진임박'
        WHEN spent_budget / {{ weekly_budget_limit|5000000 }} >= 0.8 THEN '정상소진'
        WHEN spent_budget / {{ weekly_budget_limit|5000000 }} >= 0.5 THEN '소진부진'
        ELSE '소진미달'
    END as "소진_상태"
FROM advertiser_budget ab
CROSS JOIN week_params wp
WHERE spent_budget > 0
ORDER BY spent_budget DESC
LIMIT 20
```

### 7.3 활용 시나리오
- **정기 보고서**: 매주 월요일 자동으로 지난주 성과 보고
- **임시 분석**: 특정 주간의 이벤트 효과 분석
- **비교 분석**: 여러 주간의 성과를 비교하여 트렌드 파악
- **일별 추이 분석**: 주간 내 일별 패턴 파악 및 요일별 특성 분석
- **캠페인 최적화**: ROAS 기준 상위 캠페인 식별 및 예산 재분배
- **신규 광고주 모니터링**: 첫 주 성과를 통한 잠재력 평가
- **피크타임 활용**: 시간대별 성과를 통한 광고 스케줄 최적화
- **피로도 관리**: 과도 노출 사용자 식별 및 빈도 조절
- **예산 관리**: 주간 예산 소진율 모니터링 및 조정
