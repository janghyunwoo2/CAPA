# 광고 분석 월간 쿼리 가이드 (기간조회 버전)

## 7. 실제 활용 예시 (Redash 기간 파라미터 활용)

### 7.1 기간 파라미터를 활용한 개선된 쿼리
Redash의 Date 파라미터 2개(시작일/종료일)를 사용하여 기간별 분석을 지원합니다.

#### 기간 목표 달성률 (기간조회 버전)
```sql
-- 기간 목표 대비 달성률 - 기간별 목표 설정
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as days_count
),
period_actual AS (
    SELECT 
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as actual_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as actual_cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as actual_conversions,
        COUNT(DISTINCT advertiser_id) as active_advertisers,
        COUNT(DISTINCT campaign_id) as active_campaigns,
        COUNT(DISTINCT user_id) as unique_users
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
),
targets AS (
    SELECT 
        {{ daily_revenue_target|100000000 }} * days_count as target_revenue,
        {{ daily_cost_target|50000000 }} * days_count as target_cost,
        {{ daily_conversion_target|10000 }} * days_count as target_conversions
    FROM date_params
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    days_count as "기간_일수",
    ma.actual_revenue as "실적_매출",
    t.target_revenue as "목표_매출",
    ROUND(ma.actual_revenue / t.target_revenue * 100, 1) as "매출_달성률",
    ma.actual_cost as "실적_광고비",
    t.target_cost as "목표_광고비",
    ROUND(ma.actual_cost / t.target_cost * 100, 1) as "광고비_사용률",
    ma.actual_conversions as "실적_전환수",
    t.target_conversions as "목표_전환수",
    ROUND(ma.actual_conversions / t.target_conversions * 100, 1) as "전환_달성률",
    ma.active_advertisers as "활성_광고주수",
    ma.active_campaigns as "활성_캠페인수",
    ma.unique_users as "순사용자수",
    CASE WHEN ma.actual_cost > 0 
         THEN ROUND(ma.actual_revenue / ma.actual_cost, 2)
         ELSE 0 END as "기간_ROAS"
FROM period_actual ma
CROSS JOIN targets t
CROSS JOIN date_params
```

#### 기간 일별 추이 with 이동평균 (기간조회 버전)
```sql
-- 기간 일별 매출과 7일 이동평균 - 기간 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
daily_data AS (
    SELECT 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as daily_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as daily_cost,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as daily_clicks,
        COUNT(DISTINCT advertiser_id) as daily_advertisers,
        COUNT(*) as daily_impressions,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as daily_conversions
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY 1
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    date_format(date, '%Y-%m-%d') as "날짜",
    day_of_week(date) as "요일",
    daily_revenue as "일별_매출",
    daily_cost as "일별_광고비",
    daily_clicks as "일별_클릭수",
    daily_advertisers as "활성_광고주수",
    daily_impressions as "일별_노출수",
    daily_conversions as "일별_전환수",
    AVG(daily_revenue) OVER (
        ORDER BY date 
        ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    ) as "매출_7일이동평균",
    AVG(daily_cost) OVER (
        ORDER BY date 
        ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    ) as "광고비_7일이동평균",
    CASE WHEN daily_cost > 0 
         THEN ROUND(daily_revenue / daily_cost, 2)
         ELSE 0 END as "일별_ROAS"
FROM daily_data
CROSS JOIN date_params
ORDER BY date
```

#### 기간 비교 분석 (기간조회 버전)
```sql
-- 두 기간 비교 분석 - 사용자 지정 기간 간 비교
WITH period1 AS (
    SELECT 
        DATE('{{ start_date }}') as p1_start,
        DATE('{{ end_date }}') as p1_end,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as p1_days
),
period2 AS (
    SELECT 
        DATE('{{ compare_start_date }}') as p2_start,
        DATE('{{ compare_end_date }}') as p2_end,
        DATE_DIFF('day', DATE('{{ compare_start_date }}'), DATE('{{ compare_end_date }}')) + 1 as p2_days
),
period1_data AS (
    SELECT 
        '기간1' as "구분",
        date_format(p1_start, '%Y-%m-%d') || ' ~ ' || date_format(p1_end, '%Y-%m-%d') as "기간",
        p1_days as "일수",
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
        COUNT(*) as "노출수",
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수",
        COUNT(DISTINCT advertiser_id) as "광고주수",
        COUNT(DISTINCT campaign_id) as "캠페인수"
    FROM ad_combined_log_summary
    CROSS JOIN period1
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN p1_start AND p1_end
    GROUP BY 1, 2, 3
),
period2_data AS (
    SELECT 
        '기간2' as "구분",
        date_format(p2_start, '%Y-%m-%d') || ' ~ ' || date_format(p2_end, '%Y-%m-%d') as "기간",
        p2_days as "일수",
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as "매출",
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as "광고비",
        COUNT(*) as "노출수",
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as "클릭수",
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as "전환수",
        COUNT(DISTINCT advertiser_id) as "광고주수",
        COUNT(DISTINCT campaign_id) as "캠페인수"
    FROM ad_combined_log_summary
    CROSS JOIN period2
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN p2_start AND p2_end
    GROUP BY 1, 2, 3
)
SELECT 
    "구분",
    "기간",
    "일수",
    "매출",
    "광고비",
    "노출수",
    "클릭수",
    "전환수",
    "광고주수",
    "캠페인수",
    ROUND("매출" / "일수", 0) as "일평균_매출",
    ROUND("광고비" / "일수", 0) as "일평균_광고비",
    CASE WHEN "광고비" > 0 THEN ROUND("매출" / "광고비", 2) ELSE 0 END as "ROAS",
    CASE WHEN "노출수" > 0 
         THEN ROUND(CAST("클릭수" AS DOUBLE) / "노출수" * 100, 2) 
         ELSE 0 END as "CTR"
FROM (
    SELECT * FROM period1_data
    UNION ALL
    SELECT * FROM period2_data
)
```

### 7.2 Redash 파라미터 설정
- **start_date**: Date 타입, 분석 시작일
- **end_date**: Date 타입, 분석 종료일
- **compare_start_date**: Date 타입, 비교 기간 시작일
- **compare_end_date**: Date 타입, 비교 기간 종료일
- **daily_revenue_target**: Number 타입, 일일 목표 매출 (기본: 1억)
- **daily_cost_target**: Number 타입, 일일 목표 비용 (기본: 5천만)
- **daily_conversion_target**: Number 타입, 일일 목표 전환수 (기본: 1만)
- **daily_budget_limit**: Number 타입, 일일 예산 한도 (기본: 100만)

#### 기간 캠페인별 성과 분석 (기간조회 버전)
```sql
-- 기간 캠페인별 성과 TOP 10 - 기간 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
campaign_performance AS (
    SELECT 
        campaign_id,
        campaign_name,
        COUNT(DISTINCT advertiser_id) as campaign_advertisers,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as campaign_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as campaign_cost,
        COUNT(*) as campaign_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as campaign_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as campaign_conversions,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY campaign_id, campaign_name
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    CONCAT(SUBSTR(campaign_id, 1, 4), '..', SUBSTR(campaign_id, -4)) as "캠페인ID",
    campaign_name as "캠페인명",
    active_days as "활동일수",
    campaign_advertisers as "참여_광고주수",
    campaign_revenue as "캠페인_매출",
    campaign_cost as "캠페인_비용",
    campaign_impressions as "캠페인_노출수",
    campaign_clicks as "캠페인_클릭수",
    campaign_conversions as "캠페인_전환수",
    CASE WHEN campaign_cost > 0 
         THEN ROUND(campaign_revenue / campaign_cost, 2) 
         ELSE 0 END as "캠페인_ROAS",
    CASE WHEN campaign_impressions > 0 
         THEN ROUND(CAST(campaign_clicks AS DOUBLE) / campaign_impressions * 100, 2) 
         ELSE 0 END as "캠페인_CTR",
    CASE WHEN campaign_clicks > 0 
         THEN ROUND(CAST(campaign_conversions AS DOUBLE) / campaign_clicks * 100, 2) 
         ELSE 0 END as "캠페인_CVR"
FROM campaign_performance
CROSS JOIN date_params
ORDER BY campaign_revenue DESC
LIMIT 10
```

#### 기간 상위/하위 광고주 분석 (기간조회 버전)
```sql
-- 기간 상위 10개 / 하위 10개 광고주 성과 - 기간 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
advertiser_performance AS (
    SELECT 
        advertiser_id,
        advertiser_name,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as total_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as total_cost,
        COUNT(*) as total_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as total_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as total_conversions,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY advertiser_id, advertiser_name
),
ranked_advertisers AS (
    SELECT 
        *,
        ROW_NUMBER() OVER (ORDER BY total_revenue DESC) as revenue_rank,
        ROW_NUMBER() OVER (ORDER BY total_revenue ASC) as revenue_rank_reverse
    FROM advertiser_performance
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    CASE 
        WHEN revenue_rank <= 10 THEN '상위 10개'
        WHEN revenue_rank_reverse <= 10 THEN '하위 10개'
    END as "그룹",
    CONCAT(SUBSTR(advertiser_id, 1, 4), '..', SUBSTR(advertiser_id, -4)) as "광고주ID",
    advertiser_name as "광고주명",
    active_days as "활동일수",
    total_revenue as "총매출",
    total_cost as "총광고비",
    total_impressions as "총노출수",
    total_clicks as "총클릭수",
    total_conversions as "총전환수",
    CASE WHEN total_cost > 0 
         THEN ROUND(total_revenue / total_cost, 2) 
         ELSE 0 END as "ROAS",
    CASE WHEN total_impressions > 0 
         THEN ROUND(CAST(total_clicks AS DOUBLE) / total_impressions * 100, 3) 
         ELSE 0 END as "CTR"
FROM ranked_advertisers
CROSS JOIN date_params
WHERE revenue_rank <= 10 OR revenue_rank_reverse <= 10
ORDER BY revenue_rank
```

#### 기간 시간대별 평균 성과 (기간조회 버전)
```sql
-- 기간 시간대별(0-23시) 평균 성과 - 기간 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date,
        DATE_DIFF('day', DATE('{{ start_date }}'), DATE('{{ end_date }}')) + 1 as days_count
),
hourly_data AS (
    SELECT 
        hour,
        COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                        LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                        LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as active_days,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) / COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                                                                                   LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                                                                                   LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as avg_revenue_per_hour,
        (SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END)) / COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                                                                                                               LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                                                                                                               LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as avg_cost_per_hour,
        CAST(SUM(CASE WHEN is_click THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                                                                                     LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                                                                                     LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as avg_clicks_per_hour,
        CAST(SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) AS DOUBLE) / COUNT(DISTINCT date_parse(concat(CAST(year AS VARCHAR), '-', 
                                                                                                          LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                                                                                                          LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')) as avg_conversions_per_hour,
        COUNT(*) as total_ads_shown
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY hour
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    days_count as "기간_일수",
    hour as "시간대",
    active_days as "활동일수",
    ROUND(avg_revenue_per_hour, 0) as "평균_시간당_매출",
    ROUND(avg_cost_per_hour, 0) as "평균_시간당_광고비",
    ROUND(avg_clicks_per_hour, 2) as "평균_시간당_클릭수",
    ROUND(avg_conversions_per_hour, 4) as "평균_시간당_전환수",
    total_ads_shown as "총_광고_노출수",
    CASE WHEN avg_cost_per_hour > 0 
         THEN ROUND(avg_revenue_per_hour / avg_cost_per_hour, 2) 
         ELSE 0 END as "평균_ROAS"
FROM hourly_data
CROSS JOIN date_params
ORDER BY hour
```

#### 기간 전환 퍼널 분석 (기간조회 버전)
```sql
-- 기간 전환 퍼널 분석 - 기간 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
funnel_stats AS (
    SELECT 
        COUNT(*) as total_impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as total_clicks,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as total_conversions,
        COUNT(DISTINCT user_id) as unique_users,
        COUNT(DISTINCT CASE WHEN is_click THEN user_id END) as users_clicked,
        COUNT(DISTINCT CASE WHEN is_conversion THEN user_id END) as users_converted,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as total_revenue,
        SUM(cost_per_impression) + SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as total_cost
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    '1. 노출' as "퍼널_단계",
    total_impressions as "이벤트_수",
    unique_users as "유니크_사용자",
    100.0 as "전체_대비_비율",
    100.0 as "이전_단계_대비_비율",
    0 as "단계_수익"
FROM funnel_stats
CROSS JOIN date_params
UNION ALL
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d'),
    '2. 클릭',
    total_clicks,
    users_clicked,
    ROUND(100.0 * total_clicks / total_impressions, 2),
    ROUND(100.0 * total_clicks / total_impressions, 2),
    0
FROM funnel_stats
CROSS JOIN date_params
UNION ALL
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d'),
    '3. 전환',
    total_conversions,
    users_converted,
    ROUND(100.0 * total_conversions / total_impressions, 2),
    ROUND(100.0 * total_conversions / total_clicks, 2),
    total_revenue
FROM funnel_stats
CROSS JOIN date_params
```

#### 기간 일별 주요 지표 트렌드 (기간조회 버전)
```sql
-- 기간 일별 주요 지표 트렌드 - 기간 파라미터 사용
WITH date_params AS (
    SELECT 
        DATE('{{ start_date }}') as start_date,
        DATE('{{ end_date }}') as end_date
),
daily_metrics AS (
    SELECT 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d') as date,
        COUNT(*) as impressions,
        SUM(CASE WHEN is_click THEN 1 ELSE 0 END) as clicks,
        SUM(cost_per_impression) as total_cpm_cost,
        SUM(CASE WHEN is_click THEN cost_per_click ELSE 0 END) as total_cpc_cost,
        SUM(CASE WHEN is_conversion THEN 1 ELSE 0 END) as conversions,
        SUM(CASE WHEN is_conversion THEN conversion_value ELSE 0 END) as revenue
    FROM ad_combined_log_summary
    CROSS JOIN date_params
    WHERE 
        date_parse(concat(CAST(year AS VARCHAR), '-', 
                         LPAD(CAST(month AS VARCHAR), 2, '0'), '-', 
                         LPAD(CAST(day AS VARCHAR), 2, '0')), '%Y-%m-%d')
        BETWEEN start_date AND end_date
    GROUP BY 1
)
SELECT 
    date_format(start_date, '%Y-%m-%d') || ' ~ ' || date_format(end_date, '%Y-%m-%d') as "조회기간",
    date_format(date, '%Y-%m-%d') as "날짜",
    ROUND((total_cpm_cost / impressions) * 1000, 0) as "CPM",
    CASE WHEN clicks > 0 
         THEN ROUND(total_cpc_cost / clicks, 0) 
         ELSE 0 END as "CPC",
    CASE WHEN conversions > 0 
         THEN ROUND((total_cpm_cost + total_cpc_cost) / conversions, 0) 
         ELSE 0 END as "CPA",
    CASE WHEN impressions > 0 
         THEN ROUND(CAST(clicks AS DOUBLE) / impressions * 100, 2) 
         ELSE 0 END as "CTR",
    CASE WHEN clicks > 0 
         THEN ROUND(CAST(conversions AS DOUBLE) / clicks * 100, 2) 
         ELSE 0 END as "CVR",
    CASE WHEN (total_cpm_cost + total_cpc_cost) > 0 
         THEN ROUND(revenue / (total_cpm_cost + total_cpc_cost), 2) 
         ELSE 0 END as "ROAS",
    impressions as "노출수",
    clicks as "클릭수",
    conversions as "전환수",
    revenue as "매출"
FROM daily_metrics
CROSS JOIN date_params
ORDER BY date
```

### 7.3 활용 시나리오
- **자유로운 기간 설정**: 1개월, 분기, 반기, 연간 등 원하는 기간으로 분석
- **연도별 비교**: 같은 기간의 연도별 성과 비교 (예: 2024년 1분기 vs 2025년 1분기)
- **계절성 분석**: 특정 계절이나 시즌의 성과 패턴 분석
- **캠페인 효과 측정**: 대규모 캠페인 전후 기간 비교
- **목표 관리**: 분기별, 반기별 목표 달성률 추적
- **장기 트렌드**: 6개월, 1년 등 장기간의 성과 추이 분석
- **예산 계획**: 과거 데이터 기반 미래 예산 수립
- **ROI 분석**: 장기 투자 대비 수익률 계산
